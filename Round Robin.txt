#include <stdio.h>

int main() {
    int i, limit, total = 0, x, counter = 0, time_quantum;
    int wait_time = 0, turnaround_time = 0, arrival_time[10], burst_time[10], temp[10];
    float avg_wait_time, avg_tat;

    printf("Enter the Total Number Of Process: ");
    scanf("%d", &limit);

    x = limit;

    printf("\nEnter Details Of %d Processes", limit);

    for (i = 0; i < limit; i++) {
        printf("\nEnter Arrival Time of Process %d: ", i + 1);
        scanf("%d", &arrival_time[i]);
        printf("Enter Burst Time of Process %d: ", i + 1);
        scanf("%d", &burst_time[i]);
        temp[i] = burst_time[i];
    }

    printf("\nEnter Time Quantum: ");
    scanf("%d", &time_quantum);

    printf("\nProcess\tAT\tBT\tTAT\tWT\n");

    for (total = 0, i = 0; x != 0;) {
        if (temp[i] <= time_quantum && temp[i] > 0) {
            total = total + temp[i];
            temp[i] = 0;
            counter = 1;
        } else if (temp[i] > 0) {
            temp[i] = temp[i] - time_quantum;
            total = total + time_quantum;
        }

        if (temp[i] == 0 && counter == 1) {
            x--;
            printf("\nP%d\t%d\t%d\t%d\t%d", i + 1, arrival_time[i], burst_time[i], total - arrival_time[i], total - arrival_time[i] - burst_time[i]);
            wait_time = wait_time + total - arrival_time[i] - burst_time[i];
            turnaround_time = turnaround_time + total - arrival_time[i];
            counter = 0;
        }

        if (i == limit - 1)
            i = 0;
        else if (arrival_time[i + 1] <= total)
            i++;
        else
            i = 0;
    }

    avg_wait_time = wait_time * 1.0 / limit;
    avg_tat = turnaround_time * 1.0 / limit;

    printf("\n\nAverage Waiting Time: %f", avg_wait_time);
    printf("\nAverage Turnaround Time: %f\n", avg_tat);

    return 0;
}
