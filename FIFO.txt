#include <stdio.h>

void printFrames(int frames[], int capacity, int count) {
    for (int i = 0; i < capacity; i++) {
        if (i < count)
            printf("%d ", frames[i]);
        else
            printf("- ");
    }
    printf("\n");
}

int search(int frames[], int count, int page) {
    for (int i = 0; i < count; i++) {
        if (frames[i] == page)
            return 1;
    }
    return 0;
}

void fifo(int pages[], int n, int capacity) {
    int frames[capacity];
    int count = 0, front = 0;
    int faults = 0, hits = 0;

    printf("\nRunning FIFO with %d frames\n", capacity);

    for (int i = 0; i < n; i++) {
        int page = pages[i];

        if (!search(frames, count, page)) {
            faults++;
            if (count < capacity) {
                frames[count++] = page;
            } else {
                frames[front] = page;
                front = (front + 1) % capacity;
            }
        } else {
            hits++;
        }

        printf("After inserting page %d: ", page);
        printFrames(frames, capacity, count);
    }

    printf("\nTotal Page Faults = %d\n", faults);
    printf("Total Page Hits = %d\n", hits);
    printf("Fault Ratio = %.2f\n", (float)faults / n);
    printf("Hit Ratio = %.2f\n\n", (float)hits / n);
}

int main() {
    int n, pages[100];

    printf("Enter number of pages: ");
    scanf("%d", &n);

    printf("Enter the page reference string:\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &pages[i]);
    }

    printf("\nRunning FIFO for 3 frames:\n");
    fifo(pages, n, 3);

    printf("Running FIFO for 4 frames:\n");
    fifo(pages, n, 4);

    return 0;
}
