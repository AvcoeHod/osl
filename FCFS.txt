#include <stdio.h>

int main() {
    int n, frames, pages[50], frame[10];
    int i, j, k = 0, flag, page_faults = 0;

    printf("Enter number of pages: ");
    scanf("%d", &n);

    printf("Enter page reference string: ");
    for (i = 0; i < n; i++)
        scanf("%d", &pages[i]);

    printf("Enter number of frames: ");
    scanf("%d", &frames);

    for (i = 0; i < frames; i++)
        frame[i] = -1;

    printf("\nPage Reference String\tFrames\t\tPage Fault\n");

    for (i = 0; i < n; i++) {
        flag = 0;
        for (j = 0; j < frames; j++) {
            if (frame[j] == pages[i]) {
                flag = 1;
                break;
            }
        }

        if (flag == 0) {
            frame[k] = pages[i];
            k = (k + 1) % frames;
            page_faults++;
            printf("%d\t\t\t", pages[i]);
            for (j = 0; j < frames; j++) {
                if (frame[j] != -1)
                    printf("%d ", frame[j]);
                else
                    printf("- ");
            }
            printf("\t\tYes\n");
        } else {
            printf("%d\t\t\t", pages[i]);
            for (j = 0; j < frames; j++) {
                if (frame[j] != -1)
                    printf("%d ", frame[j]);
                else
                    printf("- ");
            }
            printf("\t\tNo\n");
        }
    }

    printf("\nTotal Page Faults = %d\n", page_faults);
    printf("Total Page Hits = %d\n", n - page_faults);
    printf("Page Fault Ratio = %.2f\n", (float)page_faults / n);

    return 0;
}
